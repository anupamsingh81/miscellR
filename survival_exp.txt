
    model
      {
      # prior
      lambda ~ dgamma(0.01, 0.01)
      
      # likelihood
      for(t in 1:N)
      {
      t.to.death[t] ~ dexp(lambda)
      }
      # mean time to death
      mu <- 1/lambda
      
      # predictions
      for(i in 1:new.N)
      {
      S.t[i] <- exp(-new.t[i]/mu)
      }
      }
